def query_climate_database(db_file, country, year_begin, year_end, month):
    
    # Open a new connection to the specified database file
    conn = sqlite3.connect(db_file)
    
    # SQL command
    cmd = f"""
    SELECT 
        S.NAME, 
        S.LATITUDE, 
        S.LONGITUDE, 
        C.NAME as Country, 
        T.Year, 
        T.Month, 
        T.Temp
    FROM 
        stations S
    INNER JOIN 
        temperatures T ON T.ID = S.ID
    INNER JOIN 
        countries C ON S.ID LIKE C.'FIPS 10-4' || '%'
    WHERE 
        C.NAME = ? AND 
        T.Year BETWEEN ? AND ? AND 
        T.Month = ? AND
        T.Temp IS NOT NULL
    """
    
    # Execute the query and store the results in a pandas DataFrame
    df = pd.read_sql_query(cmd, conn, params=(country, year_begin, year_end, month))
    
    # Close the database connection
    conn.close()
    
    return df

3. Write a Geographic Scatter Function for Yearly Temperature Increases

In this part, I will write a function to create visualizations that address the following question: How does the average yearly change in temperature vary within a given country?


import plotly.express as px
from sklearn.linear_model import LinearRegression

def compute_linear_regression_coefficient(x, y):
    # Compute the coefficient (slope) of a linear regression model
    model = LinearRegression()
    model.fit(x.reshape(-1, 1), y)
    return model.coef_[0]

def temperature_coefficient_plot(db_file, country, year_begin, year_end, month, min_obs, **kwargs):
    # Connect to the database
    conn = sqlite3.connect(db_file)
    
    # Fetch the data
    df = pd.read_sql_query(
        f"""
        SELECT 
            S.NAME, 
            S.LATITUDE, 
            S.LONGITUDE, 
            T.YEAR, 
            T.VALUE{month} as Temp
        FROM 
            stations S
        JOIN 
            temperatures T ON T.ID = S.ID
        JOIN 
            countries C ON SUBSTR(T.ID, 1, 2) = C.'FIPS 10-4'
        WHERE 
            C.NAME = ? AND 
            T.YEAR BETWEEN ? AND ? AND 
            T.VALUE{month} IS NOT NULL
        """,
        conn, params=(country, year_begin, year_end))

    # Close the connection
    conn.close()
    
    # Ensure that we only consider stations with at least `min_obs` years of data
    counts = df.groupby('NAME').size()
    valid_stations = counts[counts >= min_obs].index
    df = df[df['NAME'].isin(valid_stations)]

    # Calculate the yearly temperature change coefficient for each station
    df['Year'] = df['Year'].astype(int)
    df['Temp'] = df['Temp'].astype(float) / 100  # assuming the temperature is in centi-degrees
    coef_df = df.groupby('NAME').apply(lambda x: compute_linear_regression_coefficient(x['Year'].values, x['Temp'].values))

    # Prepare the dataframe for plotting
    plot_df = df[['NAME', 'LATITUDE', 'LONGITUDE']].drop_duplicates()
    plot_df = plot_df.set_index('NAME').join(coef_df.rename('Temp_Change')).reset_index()
    
    # Create the scatter plot
    fig = px.scatter_mapbox(
        plot_df,
        lat="LATITUDE",
        lon="LONGITUDE",
        color="Temp_Change",
        hover_name="NAME",
        hover_data={"Temp_Change": ":.2f"},
        center={"lat": plot_df['LATITUDE'].mean(), "lon": plot_df['LONGITUDE'].mean()},
        **kwargs
    )

    # Add colorbar title and center the color at 0
    fig.update_layout(
        coloraxis_colorbar=dict(
            title="Yearly Temperature Change(°C/year)",
            ticks="outside",
            tickvals=[coef_df.min(), 0, coef_df.max()],
            ticktext=[f"{coef_df.min():.2f}", "0", f"{coef_df.max():.2f}"]
        ),
        title=f"Estimate of Yearly Change in Temperature of Stations in {country} in month {month} from {year_begin} to {year_end}"
    )

    return fig

color_map = px.colors.diverging.RdGy_r  # choose a colormap

import plotly.express as px

def query_annual_temp_distribution(db_file, country):
    conn = sqlite3.connect(db_file)
    query = """
    SELECT S.NAME, S.LATITUDE, S.LONGITUDE, AVG((T.VALUE1 + T.VALUE2 + T.VALUE3 + T.VALUE4 + T.VALUE5 + T.VALUE6 + T.VALUE7 + T.VALUE8 + T.VALUE9 + T.VALUE10 + T.VALUE11 + T.VALUE12)/12) as AnnualAvgTemp
    FROM stations S
    JOIN temperatures T ON T.ID = S.ID
    JOIN countries C ON SUBSTR(T.ID, 1, 2) = C.'FIPS 10-4'
    WHERE C.NAME = ?
    GROUP BY S.NAME, S.LATITUDE, S.LONGITUDE
    """
    df = pd.read_sql_query(query, conn, params=(country,))
    conn.close()
    return df
def visualize_annual_avg_temp(df, country):
    # Creating a bar chart using Plotly
    fig = px.bar(df, x='NAME', y='AnnualAvgTemp', title=f'Annual Average Temperature at Each Station in {country}',
                 labels={'NAME': 'Station Name', 'AnnualAvgTemp': 'Annual Average Temperature (°C)'})
    
    # Enhancing the layout
    fig.update_layout(xaxis_title='Station Name', 
                      yaxis_title='Annual Average Temperature (°C)', 
                      xaxis_tickangle=-45)

    # Show the figure
    return fig
